Containers

There are 4 different types of containers; div, fieldset, details, and tabs. Each allows you to wrap form elements inside a specific elements, allowing you to create various form structures. When dealing with contianers, the layout requires you to specify an array of items which can be rendered inside the container.

Div

The div is the simplest contianer. It allow you to create a div in which you can style as you wish.

layout.json
[{
    "type": "div",
    "items": [{
        "type": "text",
        "title": "Text 1"
    }, {
        "type": "text",
        "title": "Text 2"
    }, {
        "type": "text",
        "title": "Text 3"
    }]
}]

Fieldset

A fieldset container acts much the same as div, except it uses a <fieldset> element instead of a <div>. This has the added benefit that you can easily disable all form elements inside the fieldset simply by marking the container as disabled.

layout.json
[{
    "type": "fieldset",
    "items": [...],
    "disabled": true
}]

Details

A details container uses a <details> element allowing you to expand/collapse a set of form elements.

layout.json
[{
    "type": "details",
    "items": [...]
}]

Tabs

With a tabs container, you supply a list of LayoutItems of type tab. Each tab should have a title which is used to display the tab text. By default, the first tab will be selected. You can optional specify which tab to select by adding selectedIndex to the tabs options.

layout.json
[{
    "type": "tabs",
    "selectedIndex": 1,
    "items": [{
        "type": "tab",
        "title": "Tab 1",
        "items": [...]
    }, {
        "type": "tab",
        "title": "Tab 2",
        "items": [...]
    }, {
        "type": "tab",
        "title": "Tab 3",
        "items": [...]
    }]
}]

The above layout will render a container with 3 tabs; 'Tab 1', 'Tab 2', and 'Tab 3'. 'Tab 2' will be selected by default.

import { Component, OnInit } from '@angular/core';

import { Widget } from '../core/decorators/widget.decorator';
import { LayoutNode } from '../core/models/layout-node';
import { ElementDataStorageService } from '../core/services/element-data-storage.service';
import { AbstractWidget } from '../core/widget/widget';

import { JsonSchemaFormService } from '../form/services/json-schema-form.service';

/** Displays a div, fieldset, details or tabs/tab element and renders children inside it */
@Widget({
    types: ['div', 'fieldset', 'details', 'tabs', 'tab']
})
@Component({
    selector: 'jsf-container',
    styleUrls: ['./container.component.scss'],
    templateUrl: './container.component.html'
})
export class ContainerComponent extends AbstractWidget implements OnInit {
    public selectedLayout: LayoutNode;

    constructor(jsf: JsonSchemaFormService, elementDataStorage: ElementDataStorageService) {
        super(jsf, elementDataStorage);
    }

    ngOnInit() {
        super.ngOnInit();
        this.selectedLayout = this.layoutNode.type === 'tabs'
            ? this.layoutNode.items[this.getSelectedIndex()]
            : undefined;
    }

    trackByFn(index) {
        return index;
    }

    onTabClick(layout: LayoutNode): void {
        this.selectedLayout = layout;
    }

    private getSelectedIndex(): number {
        const index = Number.isInteger(this.layoutNode.options.selectedIndex)
            ? this.layoutNode.options.selectedIndex
            : 0;

        return Math.min(Math.max(index, 0), this.layoutNode.items.length - 1);
    }

}

    <ng-container *ngSwitchCase="'div'">
        <div
            [attr.accesskey]="options.accesskey"
            [attr.class]="options.htmlClass"
            [attr.tabindex]="options.tabindex"
            [attr.title]="options.description"
            [ngStyle]="options.style"
            [id]="layoutNode.id"
        >
            <h3 [hidden]="!options.title">{{ options.title }}</h3>
            <ng-container *ngTemplateOutlet="widgets"></ng-container>
        </div>
    </ng-container>
    <ng-container *ngSwitchCase="'fieldset'">
        <fieldset
            [attr.accesskey]="options.accesskey"
            [attr.class]="options.htmlClass"
            [attr.tabindex]="options.tabindex"
            [attr.title]="options.description"
            [ngStyle]="options.style"
            [id]="layoutNode.id"
            [name]="controlName"
        >
            <legend>{{ options.title }}</legend>
            <ng-container *ngTemplateOutlet="widgets"></ng-container>
        </fieldset>
    </ng-container>
    <ng-container *ngSwitchCase="'details'">
        <details
            [attr.accesskey]="options.accesskey"
            [attr.class]="options.htmlClass"
            [attr.tabindex]="options.tabindex"
            [attr.title]="options.description"
            [attr.open]="options.open ? 'open' : null"
            [ngStyle]="options.style"
            [id]="layoutNode.id"
        >
            <summary>{{ options.title }}</summary>
            <ng-container *ngTemplateOutlet="widgets"></ng-container>
        </details>
    </ng-container>
    <ng-container *ngSwitchCase="'tabs'">
        <nav
            role="tablist"
            [attr.accesskey]="options.accesskey"
            [attr.class]="options.htmlClass"
            [attr.tabindex]="options.tabindex"
            [attr.title]="options.description"
            [ngStyle]="options.style"
            [id]="layoutNode.id"
        >
            <button
                *ngFor="let _layoutNode of layoutNode.items; trackBy: trackByFn;"
                role="tab"
                type="button"
                [attr.accesskey]="_layoutNode.options.accesskey"
                [attr.class]="_layoutNode.options.htmlClass"
                [attr.tabindex]="_layoutNode.options.tabindex"
                [attr.title]="_layoutNode.options.description"
                [class.active]="selectedLayout === _layoutNode"
                [ngStyle]="_layoutNode.options.style"
                [id]="_layoutNode.id"
                (click)="onTabClick(_layoutNode)"
                >{{ _layoutNode.options.title }}
            </button>
        </nav>
        <ng-container *ngTemplateOutlet="widgets"></ng-container>
    </ng-container>
    <ng-container *ngSwitchCase="'tab'">
        <ng-container *ngTemplateOutlet="widgets"></ng-container>
    </ng-container>
</ng-container>
<ng-template #widgets>
    <jsf-select-widget
        *ngFor="let _layoutNode of layoutNode.items; trackBy: trackByFn;"
        [layoutNode]="_layoutNode"
        [hidden]="selectedLayout && selectedLayout !== _layoutNode"
    ></jsf-select-widget>
</ng-template>
